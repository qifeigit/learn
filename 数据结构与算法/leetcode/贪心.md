#### [621. 任务调度器](https://leetcode-cn.com/problems/task-scheduler/)

```java
public int leastInterval(char[] tasks, int n) {
  int[] map = new int[26];
  int len = tasks.length;
  int time = 0;
  for(int i=0;i<len;i++){
    map[tasks[i] - 'A']++;
  }
  Arrays.sort(map);
   //当数组中没有任何元素时
  while(map[25] != 0){
    int i = 0;
    while(i <= n){
     
      if(map[25] == 0){
        break;
      }
      //每次都从最大的开始遍历
      if( 25 - i >= 0 && map[25 - i]>=1){
        map[25-i]--;
      }
      i++;
      time++;
    }
    Arrays.sort(map);
  }
  return time;
}
```
